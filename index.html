Issue 16: Tiny DSL (hmm, implicits is a grenade in monkey's hands)

class Theme(val name: String, val phrases: List[String]) {
  def and(keyPhrase: String) = new Theme(name, phrases :+ keyPhrase)
}
object ClassificatorExtensions{
  implicit def stringWrapper(text: String) = new Theme(text, List())
}
import ClassificatorExtensions._
object StandardThemes {
  def getAll:List[Theme] = {
    List(
      "Linux" and "Linux" and "Ubuntu",
      "Javascript" and "Javascript" and "Requirejs",
      "Google" and "Google" and "Android")
  }
}
or so:
object ClassificatorExtensions{
  implicit class ThemeExtension(val theme: Theme){
    def and(keyPhrase: String) = {
      new Theme(theme.name, theme.phrases :+ keyPhrase)
    }
  }
  implicit class StringExtension(val themeName: String){
    def is(keyPhrase: String) = {
      new Theme(themeName, List(keyPhrase))
    }
  }
}
    "Linux" is "Linux" and "Ubuntu",
    "Javascript" is "Javascript" and "Requirejs",
    "Google" is "Google" and "Android"
